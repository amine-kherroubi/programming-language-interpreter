{
    let int a = 10
    let float b = 3.2
    keep bool flag = true

    proc doNothing() {
        let int x = 1
        let float y = 2.0
    }

    func multiply(x: int, y: float) -> float {
        let float result = x * y
        give result
    }

    exec doNothing()

    let float result = multiply(a, b)
}
